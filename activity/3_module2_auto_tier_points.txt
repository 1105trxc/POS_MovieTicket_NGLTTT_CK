:Hệ thống lấy thông tin Customer từ DB;

  :KHÔNG tính điểm thưởng;
  :EarnedPoints = 0;
else (Sai (Cho phép tích điểm))
  :EarnedPoints = FinalAmount * 5%;
  :Cộng dồn điểm: Customer.RewardPoints += EarnedPoints;
  :Ghi Log vào PointHistory (TransactionType="EARN");
endif

:Cộng dồn Tổng chi tiêu: Customer.TotalSpent += FinalAmount;


if (TotalSpent >= 5.000.000đ ?) then (Vượt hạng Gold)
  :Đổi Hạng Customer -> GOLD;
else (Dưới Gold)
  if (TotalSpent >= 2.000.000đ ?) then (Vượt hạng Silver)
     :Đổi Hạng Customer -> SILVER;
  else (Dưới Silver)
     :Mặc định Hạng -> REGULAR;
  endif
endif

:Lưu (Update) Customer vào Database;
:Phản hồi thông tin Điểm và Hạng cập nhật;
