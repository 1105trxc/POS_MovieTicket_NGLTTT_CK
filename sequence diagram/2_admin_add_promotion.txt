title Sequence Diagram: Quản lý Khuyến Mãi - Tạo mới

actor Admin as admin
participant "PromotionManagementPanel" as view
participant "PromotionController" as ctl
participant "PromotionServiceImpl" as svc
participant "AuditLogServiceImpl" as audit
participant "PromotionRepository" as repo
database "Database" as db

admin -> view: Nhập dữ liệu form và nhấn "Lưu"
view -> ctl: addPromotion(promotion)
ctl -> svc: createPromotion(promotion)
svc -> repo: save(promotion)
repo -> db: INSERT INTO Promotion
db --> repo: Success
svc -> audit: logAction("CREATE", "Promotion", ...)
audit -> db: INSERT INTO AuditLog
db --> audit: Success
svc --> ctl: (void)
ctl --> view: Hiển thị thông báo thành công
view --> admin: Cập nhật lại JTable
@enduml
