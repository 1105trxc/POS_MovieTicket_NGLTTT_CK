@startuml
title Sequence Diagram: (Module 3) Đăng nhập và Phân Quyền (Auth)

actor User as usr
participant "LoginFrame" as view
participant "UserController" as ctl
participant "UserServiceImpl" as svc
participant "UserRepository" as repo
participant "UserSessionContext" as session
database "Database" as db

usr -> view: Nhập Username/Password + Đăng nhập
view -> ctl: validateLogin(username, password)
ctl -> svc: checkLoginCredentials(...)
svc -> repo: findByUsername(username)
repo -> db: SELECT FROM User JOIN Role 
db --> repo: User Entity
repo --> svc: User Entity
svc -> svc: Compare Password (Băm MD5/Bcrypt)
svc --> ctl: Nếu Sai -> Báo Lỗi \nNếu Đúng -> User Entity

ctl -> session: setCurrentUser(User Data)
session --> ctl: Set Tĩnh Mọi Nơi
ctl --> view: Thành công

view -> session: getUserRole() (isAdmin?)
session --> view: True / False
alt Là ADMIN
  view -> view: Mở MainFrame đầy đủ Module Staff, Audit, Revenue...
else Là STAFF
  view -> view: Trỏ thẳng vô View Bán Vé (Ticket Component), Khoá nút...
end
view --> usr: App Khởi chạy Xong
@enduml
