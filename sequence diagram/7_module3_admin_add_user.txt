title Sequence Diagram: (Module 3) Admin tạo Tài khoản Nhân Viên & Ghi Log

actor Admin as admin
participant "UserManagementPanel" as view
participant "UserController" as ctl
participant "UserServiceImpl" as svc
participant "AuditLogServiceImpl" as audit
participant "UserRepository" as repo
database "Database" as db

admin -> view: Điền form tạo Staff (Kèm Password) + Bấm Lưu
view -> ctl: createUser(User)
ctl -> svc: createUser(User)
svc -> svc: Hash (Mã hoá) Password
svc -> repo: Tương tác save(User)
repo -> db: INSERT INTO [User]
db --> repo: Thành công
repo --> svc: (Void)

svc -> audit: logAction("CREATE", "User", "Role: STAFF", ...)
audit -> db: INSERT INTO AuditLog
db --> audit: (Thành công)
audit --> svc: (Void)

svc --> ctl: (Void)
ctl --> view: Refresh Data Table
view --> admin: Thấy Account Mới
